    <!-- Chart Configuration and Dynamic Interaction Script -->
    <script>
        // Mathematical Constants for Logistic Model
        const L = 80;    // Carrying Capacity (Max radius in mm)
        const k = 0.65;  // Growth Rate constant
        const x0 = 6;    // Midpoint day of maximum growth rate

        // Logistic Function f(x)
        function f(x) {
            return L / (1 + Math.exp(-k * (x - x0)));
        }

        // Derivative Function f'(x)
        function f_prime(x) {
            const fx = f(x);
            return k * fx * (1 - (fx / L));
        }

        // Generate dense data points for seamless tracking
        const labels = [];
        const growthData = [];
        const steps = 140; // 0 to 14 days broken into steps of 0.1 days
        
        for (let i = 0; i <= steps; i++) {
            const day = i / 10;
            labels.push(day.toFixed(1));
            growthData.push(f(day));
        }

        // Initialize Chart.js configuration
        const ctx = document.getElementById('myceliumChart').getContext('2d');
        
        // Gradient fill for area under the curve
        const chartGradient = ctx.createLinearGradient(0, 0, 0, 300);
        chartGradient.addColorStop(0, 'rgba(16, 185, 129, 0.4)');
        chartGradient.addColorStop(1, 'rgba(16, 185, 129, 0.0)');

        const myceliumChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [{
                    label: 'Mycelium Colony Radius f(x) (mm)',
                    data: growthData,
                    borderColor: '#059669', // Emerald 600
                    borderWidth: 3,
                    pointRadius: 0, // Hidden until hover/click
                    pointHoverRadius: 6,
                    pointBackgroundColor: '#dc2626', // Red 600 pointer indicator
                    pointHoverBorderColor: '#ffffff',
                    pointHoverBorderWidth: 2,
                    fill: true,
                    backgroundColor: chartGradient,
                    tension: 0.2
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: {
                        position: 'top',
                        labels: { font: { weight: '500', size: 12 } }
                    },
                    tooltip: {
                        enabled: true,
                        callbacks: {
                            label: function(context) {
                                return ` Radius: ${context.parsed.y.toFixed(2)} mm`;
                            }
                        }
                    }
                },
                scales: {
                    x: {
                        title: { display: true, text: 'Incubation Period (Days)', font: { weight: '600' } },
                        grid: { display: false }
                    },
                    y: {
                        title: { display: true, text: 'Colony Expansion (mm)', font: { weight: '600' } },
                        min: 0,
                        max: 90,
                        ticks: { stepSize: 10 }
                    }
                },
                // Trigger action when the user clicks anywhere on the graph area
                onClick: (e, elements) => {
                    if (elements.length > 0) {
                        const index = elements[0].index;
                        const selectedDay = parseFloat(labels[index]);
                        updateMetrics(selectedDay);
                        
                        // Highlight the selected point visually
                        myceliumChart.setActiveElements([{ datasetIndex: 0, index: index }]);
                        myceliumChart.update();
                    }
                },
                animation: {
                    duration: 1500,
                    easing: 'easeOutQuart'
                }
            }
        });

        // Function to process values and update the DOM elements dynamically
        function updateMetrics(day) {
            const currentRadius = f(day);
            const currentDerivative = f_prime(day);

            // Update numerical text contents
            document.getElementById('selectedDayBadge').innerText = `Day ${day.toFixed(1)}`;
            document.getElementById('outRadius').innerText = currentRadius.toFixed(2);
            document.getElementById('outDerivative').innerText = currentDerivative.toFixed(2);

            // Generate contextual layman descriptions based on the calculated values
            let description = "";
            if (day < 3.5) {
                description = `The growth is in its early lag phase. The mycelium is adapting to the rubberwood sawdust substrate block. It expands at a gentle speed of ${currentDerivative.toFixed(2)} mm/day.`;
            } else if (day >= 3.5 && day <= 8.5) {
                description = `The growth is in its optimal log phase. The derivative value shows a highly aggressive expansion speed of ${currentDerivative.toFixed(2)} mm/day. Keep the farm humidity up to standard right now!`;
            } else {
                description = `The growth rate is declining (deceleration phase). The mycelium has almost entirely bound the substrate pouch block, meaning nutrients are depleting, and the radius is reaching its structural capacity limit.`;
            }

            document.getElementById('laymanText').innerText = description;

            // Flash effect for interactivity confirmation
            const card = document.getElementById('resultCard');
            card.classList.remove('scale-100');
            card.classList.add('scale-[1.01]', 'border-emerald-500');
            setTimeout(() => {
                card.classList.remove('scale-[1.01]', 'border-emerald-500');
                card.classList.add('scale-100');
            }, 150);
        }

        // Set default values matching initial midpoint loading view state
        window.onload = () => {
            updateMetrics(6.0);
        };
    </script>
